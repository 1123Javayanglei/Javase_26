    public static void judgmentCompletionOfSeven(int num){
        int[] array =new int[100];
        for (int i = 1,j=0; i <num ; i++) {
            if (num%i==0){
                array[j]=i;
                j++;
            }
        }
        int sum=0;
        for (int i = 0; i <array.length; i++) {
            sum+=array[i];
        }

        if (num==sum){
            System.out.println(" "+num+"是完数");
        }else {
            System.out.println(" "+num+"不是完数");
        }

    }
