    public static void sumOfOneToThousand(){
        /* 
         *求1到1000满足条件的所有数的和 
         * 1.偶数 %2==0 
         * 2.位数有1 Number%10==1 || Number/10%10==1 || Number/100==1 
         */
        int max=1000;
        int sum=0;
        int number=0;
        while (number<=max){
            if (((number%10==1)||(number/10%10==1) || (number/100==1))&&(number%2==0)){
                sum+=number;
            }
            number++;
        }
        System.out.println("1到1000满足条件的所有数的和="+sum);

    }
