public class TextWork {
    public static void main(String[] args) {
        int sumOfHundredToThousand=0;
        for (int i = 100; i <=1000; i++) {
            if (i%2==0||i%7==0){
                sumOfHundredToThousand+=i;
            }
        }
        System.out.println("1到1000内可以被2或7整除的所有数的和是"+sumOfHundredToThousand);

        int sumOf100To999=0;
        for (int i =100; i <=999; i++) {
            if (i%3==0&&(i%10==1||i/10%10==1||i/100==1)){
                sumOf100To999+=i;
            }
        }
        System.out.println("100到999内可以被3整除并且位数上有1的所有数的和是："+sumOf100To999);

        int sumOfThere=0,i;
        for (i=1; sumOfThere<1500 ; i++) {
            sumOfThere+=i;
        }
        System.out.println("从1开始求和 和第一次加到1500时，加到"+(i-1));

        int sumOf100To1000Cnt=0,sunOf100To1000Sum=0;
        for (int j = 100; j <=1000; j++) {
            if (j%2==0 || j%7==0)
            {
                sunOf100To1000Sum+=j;
                sumOf100To1000Cnt++;
            }
        }
        System.out.println("100到1000内可以被2或7整除的所有数的和是："+sumOf100To1000Cnt+" 100到1000内可以被2或7整除的所有数的平均数是："+(sunOf100To1000Sum/sumOf100To1000Cnt));



    }
}
